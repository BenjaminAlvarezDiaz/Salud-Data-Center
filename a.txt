-- Tabla pacientes

CREATE TABLE IF NOT EXISTS pacientes
(
    id integer primary key generated by default as identity nextval('pacientes_id_seq'::regclass),
    "Nombre" character varying(255) COLLATE pg_catalog."default",
    "Apellido" character varying(255) COLLATE pg_catalog."default",
    "Dni" integer,
    "Email" character varying(255) COLLATE pg_catalog."default",
    "Telefono" integer,
    "Telefono2" integer,
    "Sintomas" character varying(255) COLLATE pg_catalog."default",
    "Tratamiento" character varying(255) COLLATE pg_catalog."default",
    "Diagnostico" character varying(255) COLLATE pg_catalog."default",
    "Exp_Medico" character varying(255) COLLATE pg_catalog."default",
)

-- Tabla categoria

CREATE TABLE IF NOT EXISTS categoria
(
    id integer primary key generated by default as identity nextval('categoria_id_seq'::regclass),
    "Tipo" character varying(255) COLLATE pg_catalog."default",
    "Nombre" character varying(255) COLLATE pg_catalog."default",
    "createdAt" timestamp with time zone NOT NULL,
    "updatedAt" timestamp with time zone NOT NULL,
)
